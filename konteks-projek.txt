== KONTEKS PROYEK ==

Saya sedang membuat aplikasi berbasis website untuk pemantauan laporan penjualan pada Toko Madura. Aplikasi ini memiliki tiga peran pengguna, yaitu Master, Investor, dan Outlet, dengan rincian tugas sebagai berikut:

1. Tugas Master
- Mendaftarkan akun untuk setiap investor.
- Mengatur persentase potongan global. Pengaturan ini dibuat dinamis karena nilai persentase belum bersifat tetap.
- Memantau berjalannya sistem secara garis besar. Setiap Master dapat melihat daftar investor di bawah naungannya, akumulasi total omzet per investor, dan total perputaran uang dari potongan tersebut.

2. Tugas Investor
- Mendaftarkan akun untuk setiap outlet yang berada di bawah kepemilikannya.
- Melihat daftar outlet miliknya saja beserta rincian total omzet dari masing-masing outlet tersebut.
- Melihat akumulasi keseluruhan potongan dari semua outlet miliknya. Nantinya, sistem akan secara otomatis mengalkulasi bagi hasil dari total kumpulan potongan ini (pembagian antara hak Investor dan hak Outlet).

3. Tugas Outlet
- Menginput total omzet penjualan yang didapatkan dari operasional toko pada setiap akhir periode pelaporan (misalnya harian, mingguan, atau bulanan).
- Setelah penginputan, sistem akan secara otomatis menghitung nilai potongan berdasarkan persentase global yang sedang berlaku saat itu dan menyimpannya sebagai riwayat data.

== DATABASE ==

CREATE DATABASE toko_madura;
USE toko_madura;

CREATE TABLE users (
    id_users INT(10) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    nama_lengkap VARCHAR(100) NOT NULL,
    username VARCHAR(50) NOT NULL UNIQUE,
    no_hp VARCHAR(20) DEFAULT NULL,
    email VARCHAR(100) DEFAULT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role ENUM('master', 'investor', 'outlet') NOT NULL
);

CREATE TABLE investor (
    id_investor INT(10) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    id_users INT(10) UNSIGNED NOT NULL UNIQUE,
    id_master INT(10) UNSIGNED NOT NULL,
    alamat_investor TEXT DEFAULT NULL,
    persen_bagian_investor DECIMAL(5,2) NOT NULL DEFAULT 50.00,
    FOREIGN KEY (id_users) REFERENCES users(id_users) ON DELETE CASCADE,
    FOREIGN KEY (id_master) REFERENCES users(id_users) ON DELETE RESTRICT
);

CREATE TABLE outlet (
    id_outlet INT(10) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    id_users INT(10) UNSIGNED NOT NULL UNIQUE,
    id_investor INT(10) UNSIGNED NOT NULL,
    kode_outlet VARCHAR(20) NOT NULL UNIQUE,
    nama_outlet VARCHAR(100) NOT NULL,
    alamat_outlet TEXT DEFAULT NULL,
    FOREIGN KEY (id_users) REFERENCES users(id_users) ON DELETE CASCADE,
    FOREIGN KEY (id_investor) REFERENCES investor(id_investor) ON DELETE RESTRICT
);

CREATE TABLE laporan_omzet (
    id_laporan INT(10) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    id_outlet INT(10) UNSIGNED NOT NULL,
    periode_laporan DATE NOT NULL,
    omzet DECIMAL(15,2) NOT NULL DEFAULT 0.00,
    presentase_potongan DECIMAL(5,2) NOT NULL,
    nominal_potongan DECIMAL(15,2) NOT NULL,
    waktu_input DATETIME NOT NULL,
    FOREIGN KEY (id_outlet) REFERENCES outlet(id_outlet) ON DELETE RESTRICT,
    INDEX idx_laporan_periode (periode_laporan)
);

CREATE TABLE rekap_bagi_hasil (
    id_rekap INT(10) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    id_investor INT(10) UNSIGNED NOT NULL,
    periode_rekap VARCHAR(20) NOT NULL,
    akumulasi_omzet DECIMAL(15,2) NOT NULL DEFAULT 0.00,
    akumulasi_potongan DECIMAL(15,2) NOT NULL DEFAULT 0.00,
    hak_investor DECIMAL(15,2) NOT NULL DEFAULT 0.00,
    hak_outlet DECIMAL(15,2) NOT NULL DEFAULT 0.00,
    FOREIGN KEY (id_investor) REFERENCES investor(id_investor) ON DELETE RESTRICT,
    INDEX idx_rekap_periode (periode_rekap)
);

CREATE TABLE pengaturan_sistem (
    id_pengaturan INT(10) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    nama_pengaturan VARCHAR(50) NOT NULL UNIQUE,
    nilai DECIMAL(5,2) NOT NULL
);
